function generateVersionId(project: Project): string {
  const count = project.versions.length
  return `v${count + 1}`
}

### 4.2 保存时创建快照（已实现）

**位置**：`packages/author-site/src/lib/session-manager.ts` 中 `saveEditSession`

当前实现（含版本快照）：

```typescript
export function saveEditSession(
  sessionId: string,
  userId?: string,
  note?: string
): SaveResult {
  const sessionMeta = getEditSession(sessionId)
  if (!sessionMeta) return { success: false, error: 'Session not found' }

  const projectId = sessionMeta.demoId
  const projectPath = getProjectPath(projectId)
  const workspacePath = path.join(projectPath, 'workspace')
  const projectJsonPath = path.join(projectPath, 'project.json')
  const sessionPath = getSessionPath(sessionId)

  // 1. 读取项目元数据
  const project: Project = JSON.parse(
    fs.readFileSync(projectJsonPath, 'utf-8')
  )

  // 2. 生成新版本号
  const versionId = generateVersionId(project)
  const snapshotPath = getSnapshotPath(projectId, versionId)

  // 3. 【关键】创建快照：备份当前 workspace 到 snapshots
  fs.mkdirSync(path.dirname(snapshotPath), { recursive: true })
  fs.cpSync(workspacePath, snapshotPath, { recursive: true })

  // 4. 将项目工作区同步到项目基准工作区
  fs.rmSync(workspacePath, { recursive: true, force: true })
  fs.cpSync(sessionPath, workspacePath, { recursive: true })
  fs.rmSync(path.join(workspacePath, '.session.json'), { force: true })

  // 5. 记录版本信息
  const versionInfo: VersionInfo = {
    versionId,
    savedAt: Date.now(),
    savedBy: userId || 'anonymous',
    sessionId,
    snapshotPath,
    fileCount: countFiles(workspacePath),
    note,
  }

  project.versions.push(versionInfo)
  project.updatedAt = Date.now()

  // 6. 清理旧版本（保留最近 MAX_VERSIONS_KEEP 个）
  cleanupOldVersions(project)

  // 7. 保存项目元数据
  fs.writeFileSync(projectJsonPath, JSON.stringify(project, null, 2), 'utf-8')

  // 8. 清理 session
  deleteSession(sessionId)

  return {
    success: true,
    version: versionId,
    savedAt: versionInfo.savedAt,
  }
}
```

### 4.3 版本清理

**位置**：`packages/author-site/src/lib/fs-utils.ts`

```typescript
import { MAX_VERSIONS_KEEP } from '@opencode-workbench/shared'

function cleanupOldVersions(project: Project): void {
  if (project.versions.length <= MAX_VERSIONS_KEEP) return

  // 删除超出保留数量的最旧版本
  const toDelete = project.versions.slice(0, project.versions.length - MAX_VERSIONS_KEEP)
  
  for (const version of toDelete) {
    if (fs.existsSync(version.snapshotPath)) {
      fs.rmSync(version.snapshotPath, { recursive: true, force: true })
    }
  }

  // 保留最近 MAX_VERSIONS_KEEP 个
  project.versions = project.versions.slice(-MAX_VERSIONS_KEEP)
}
```

### 4.4 版本历史查询

**位置**：`packages/author-site/src/lib/fs-utils.ts`（新增函数）

```typescript
export function getVersionHistory(projectId: string): VersionInfo[] {
  const projectPath = getProjectPath(projectId)
  const projectJsonPath = path.join(projectPath, 'project.json')

  if (!fs.existsSync(projectJsonPath)) {
    return []
  }

  const project: Project = JSON.parse(
    fs.readFileSync(projectJsonPath, 'utf-8')
  )

  // 倒序返回：最新的在前面
  return [...project.versions].reverse()
}

export function getLatestVersion(projectId: string): VersionInfo | null {
  const versions = getVersionHistory(projectId)
  return versions.length > 0 ? versions[0] : null
}
```

### 4.5 版本恢复

**位置**：`packages/author-site/src/lib/fs-utils.ts`（新增函数）

```typescript
export function restoreVersion(
  projectId: string,
  versionId: string,
  userId?: string
): { success: boolean; newVersionId?: string; error?: string } {
  const projectPath = getProjectPath(projectId)
  const projectJsonPath = path.join(projectPath, 'project.json')
  const project: Project = JSON.parse(
    fs.readFileSync(projectJsonPath, 'utf-8')
  )

  // 查找目标版本
  const targetVersion = project.versions.find(v => v.versionId === versionId)
  if (!targetVersion) {
    return { success: false, error: `版本 ${versionId} 不存在` }
  }

  // 检查快照是否存在
  if (!fs.existsSync(targetVersion.snapshotPath)) {
    return { success: false, error: `版本快照已丢失: ${versionId}` }
  }

  const workspacePath = path.join(projectPath, 'workspace')

  // 1. 先备份当前 workspace（生成一个新版本）
  const backupVersionId = generateVersionId(project)
  const backupSnapshotPath = getSnapshotPath(projectId, backupVersionId)
  fs.cpSync(workspacePath, backupSnapshotPath, { recursive: true })

  const backupVersion: VersionInfo = {
    versionId: backupVersionId,
    savedAt: Date.now(),
    savedBy: userId || 'system',
    sessionId: `restore-from-${versionId}`,
    snapshotPath: backupSnapshotPath,
    fileCount: countFiles(workspacePath),
    note: `恢复版本前的自动备份 (基于 ${versionId})`,
  }

  project.versions.push(backupVersion)

  // 2. 用目标版本快照覆盖 workspace
  fs.rmSync(workspacePath, { recursive: true, force: true })
  fs.cpSync(targetVersion.snapshotPath, workspacePath, { recursive: true })

  // 3. 记录恢复操作（可选：作为新版本记录）
  const restoreVersionId = generateVersionId(project)
  const restoreSnapshotPath = getSnapshotPath(projectId, restoreVersionId)
  fs.cpSync(workspacePath, restoreSnapshotPath, { recursive: true })

  const restoreVersionInfo: VersionInfo = {
    versionId: restoreVersionId,
    savedAt: Date.now(),
    savedBy: userId || 'system',
    sessionId: `restore-${versionId}`,
    snapshotPath: restoreSnapshotPath,
    fileCount: countFiles(workspacePath),
    note: `恢复到版本 ${versionId}`,
  }

  project.versions.push(restoreVersionInfo)
  project.updatedAt = Date.now()

  // 4. 清理旧版本
  cleanupOldVersions(project)

  // 5. 保存项目元数据
  fs.writeFileSync(projectJsonPath, JSON.stringify(project, null, 2), 'utf-8')

  return { success: true, newVersionId: restoreVersionId }
}
```

## 九、测试要点

1. **保存流程**：保存后检查 snapshots 目录是否生成了对应版本
2. **版本号连续性**：多次保存后版本号是否为 v1, v2, v3...
3. **版本清理**：创建 55 个版本后，最旧的 5 个是否被删除

5. **恢复备份**：恢复操作前是否自动创建了当前版本的备份
6. **并发编辑**：多用户同时编辑同一项目，各自 session 是否独立
7. **过期清理**：过期 session 是否被正确清理
8. **元数据一致**：`project.json` 中的 `versions` 与实际 snapshots 是否一致

## 十、风险与注意事项

1. **磁盘空间**：每个版本都是完整副本，需监控磁盘使用
2. **大项目**：文件多的项目版本快照会占用较多空间
3. **并发保存**：同时保存可能产生竞态条件（建议加文件锁或队列）
4. **快照丢失**：手动删除 snapshots 目录会导致版本不可恢复（需加校验）
5. **回滚不可撤销**：版本恢复后，当前未保存的编辑会丢失（需提示用户）
