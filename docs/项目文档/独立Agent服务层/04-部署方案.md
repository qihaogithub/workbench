## 二、Docker 部署

### 2.1 Dockerfile

```dockerfile
# packages/agent-service/Dockerfile

FROM node:20-alpine AS builder

WORKDIR /app

RUN npm install -g pnpm

COPY package.json pnpm-lock.yaml ./
RUN pnpm install --frozen-lockfile

COPY . .
RUN pnpm build

FROM node:20-alpine AS runner

WORKDIR /app

RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 agent

COPY --from=builder /app/dist ./dist
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/package.json ./

USER agent

EXPOSE 3101

ENV NODE_ENV=production
ENV PORT=3101

CMD ["node", "dist/server.js"]
```

### 2.2 docker-compose.yml

```yaml
# docker-compose.yml

version: '3.8'

services:
  agent-service:
    build:
      context: ./packages/agent-service
      dockerfile: Dockerfile
    container_name: agent-service
    ports:
      - "3101:3101"
    environment:
      - NODE_ENV=production
      - PORT=3101
      - HOST=0.0.0.0
      - OPENCODE_SERVER_URL=http://opencode:4096
      - LOG_LEVEL=info
      - RATE_LIMIT_MAX=100
      - RATE_LIMIT_WINDOW=60000
    depends_on:
      opencode:
        condition: service_healthy
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "wget", "-q", "--spider", "http://localhost:3101/health"]
      interval: 30s
      timeout: 10s
      retries: 3
    networks:
      - agent-network

  opencode:
    image: opencode/opencode:latest
    container_name: opencode
    ports:
      - "4096:4096"
    volumes:
      - opencode-data:/data
    environment:
      - PORT=4096
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "wget", "-q", "--spider", "http://localhost:4096/global/health"]
      interval: 30s
      timeout: 10s
      retries: 3
    networks:
      - agent-network

  nginx:
    image: nginx:alpine
    container_name: nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
    depends_on:
      - agent-service
    restart: unless-stopped
    networks:
      - agent-network

volumes:
  opencode-data:

networks:
  agent-network:
    driver: bridge
```

### 2.3 Nginx 配置

```nginx
# nginx/nginx.conf

events {
    worker_connections 1024;
}

http {
    upstream agent_service {
        least_conn;
        server agent-service:3101;
    }

    upstream opencode {
        server opencode:4096;
    }

    server {
        listen 80;
        server_name _;

        location /api/agent {
            proxy_pass http://agent_service;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "upgrade";
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_read_timeout 120s;
        }

        location /health {
            proxy_pass http://agent_service;
        }

        location /opencode/ {
            proxy_pass http://opencode/;
        }
    }

    server {
        listen 443 ssl;
        server_name _;

        ssl_certificate /etc/nginx/ssl/cert.pem;
        ssl_certificate_key /etc/nginx/ssl/key.pem;

        location /api/agent {
            proxy_pass http://agent_service;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "upgrade";
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_read_timeout 120s;
        }

        location /health {
            proxy_pass http://agent_service;
        }
    }
}
```

```yaml
# k8s/agent-service-deployment.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: agent-service
  labels:
    app: agent-service
spec:
  replicas: 3
  selector:
    matchLabels:
      app: agent-service
  template:
    metadata:
      labels:
        app: agent-service
    spec:
      containers:
        - name: agent-service
          image: opencode/agent-service:latest
          ports:
            - containerPort: 3101
          env:
            - name: NODE_ENV
              value: production
            - name: PORT
              value: "3101"
            - name: OPENCODE_SERVER_URL
              value: "http://opencode-service:4096"
            - name: LOG_LEVEL
              value: info
          resources:
            requests:
              memory: "256Mi"
              cpu: "250m"
            limits:
              memory: "512Mi"
              cpu: "500m"
          livenessProbe:
            httpGet:
              path: /health
              port: 3101
            initialDelaySeconds: 10
            periodSeconds: 30
          readinessProbe:
            httpGet:
              path: /health
              port: 3101
            initialDelaySeconds: 5
            periodSeconds: 10
---
apiVersion: v1
kind: Service
metadata:
  name: agent-service
spec:
  selector:
    app: agent-service
  ports:
    - port: 80
      targetPort: 3101
  type: ClusterIP
---
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: agent-service-ingress
spec:
  rules:
    - host: agent.example.com
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: agent-service
                port:
                  number: 80
```

## 七、故障排查

### 7.1 常见问题

| 问题 | 可能原因 | 解决方案 |
|:-----|:---------|:---------|
| 服务无法启动 | 端口被占用 | 检查端口占用，修改配置 |
| 健康检查失败 | OpenCode 未运行 | 启动 OpenCode 服务 |
| 请求超时 | 网络问题 | 检查网络连接，增加超时时间 |
| 内存不足 | Agent 未释放 | 重启服务，检查内存泄漏 |

### 7.2 诊断命令

```bash
# 检查服务状态
curl http://localhost:3101/health

# 检查 OpenCode 连接
curl http://localhost:4096/global/health

# 查看容器日志
docker logs agent-service --tail 100

# 进入容器调试
docker exec -it agent-service sh

# 检查资源使用
docker stats agent-service
```

## 八、安全加固

### 8.1 网络安全

```yaml
# 限制容器网络访问
services:
  agent-service:
    networks:
      - agent-network
    # 不暴露端口到宿主机（通过 nginx 代理）
    expose:
      - "3101"
```

### 8.2 认证配置

```typescript
// 启用 API Key 认证
fastify.addHook('onRequest', async (request, reply) => {
  if (process.env.API_KEY_ENABLED === 'true') {
    const apiKey = request.headers['x-api-key'];
    if (apiKey !== process.env.API_KEY) {
      reply.code(401).send({ error: 'Unauthorized' });
    }
  }
});
```

### 8.3 HTTPS 配置

```nginx
server {
    listen 443 ssl http2;
    
    ssl_certificate /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;
}
```

## 十一、总结

本部署方案支持：

1. **Docker Compose**：单机部署，适合开发和小规模生产
2. **Kubernetes**：集群部署，适合大规模生产
3. **监控告警**：健康检查、日志、指标
4. **安全加固**：认证、HTTPS、网络隔离
5. **运维操作**：启动、停止、扩容、更新
