## 一、错误处理概述

### 1.1 设计原则

| 原则 | 说明 |
|:-----|:-----|
| 统一格式 | 所有错误使用统一格式返回 |
| 语义化 | 错误码应清晰表达错误类型 |
| 用户友好 | 错误信息应对用户友好 |
| 安全性 | 不暴露系统内部细节 |

### 1.2 错误响应格式

```typescript
{
  success: false,
  error: {
    code: ErrorCode,
    message: string,
    details?: unknown
  }
}
```

## 二、错误码定义

### 2.1 错误码常量

**文件位置**：`@opencode-workbench/shared`

```typescript
export const ErrorCode = {
  DEMO_NOT_FOUND: 'DEMO_NOT_FOUND',
  SESSION_NOT_FOUND: 'SESSION_NOT_FOUND',
  INVALID_REQUEST: 'INVALID_REQUEST',
  FILE_READ_ERROR: 'FILE_READ_ERROR',
  FILE_WRITE_ERROR: 'FILE_WRITE_ERROR',
  AI_SERVICE_ERROR: 'AI_SERVICE_ERROR',
} as const

export type ErrorCodeType = typeof ErrorCode[keyof typeof ErrorCode]
```

### 2.2 错误消息映射

```typescript
export const ERROR_MESSAGES: Record<ErrorCodeType, string> = {
  DEMO_NOT_FOUND: 'Demo 不存在',
  SESSION_NOT_FOUND: '会话不存在或已过期',
  INVALID_REQUEST: '请求参数无效',
  FILE_READ_ERROR: '文件读取失败',
  FILE_WRITE_ERROR: '文件写入失败',
  AI_SERVICE_ERROR: 'AI 服务不可用',
}
```

### 2.3 HTTP 状态码映射

| 错误码 | HTTP 状态码 |
|:-------|:------------|
| DEMO_NOT_FOUND | 404 |
| SESSION_NOT_FOUND | 404 |
| INVALID_REQUEST | 400 |
| FILE_READ_ERROR | 500 |
| FILE_WRITE_ERROR | 500 |
| AI_SERVICE_ERROR | 503 |

## 三、辅助函数

### 3.1 createApiError

**文件位置**：`src/lib/fs-utils.ts`

```typescript
export function createApiError(
  code: ErrorCodeType,
  message?: string,
  details?: unknown
) {
  return {
    success: false as const,
    error: {
      code,
      message: message || ERROR_MESSAGES[code],
      details,
    },
  }
}
```

### 3.2 使用示例

```typescript
if (!demoExists(id)) {
  return NextResponse.json(
    createApiError('DEMO_NOT_FOUND'),
    { status: 404 }
  )
}
```

## 四、错误处理模式

### 4.1 try/catch 包裹

所有 API 路由使用 try/catch 包裹：

```typescript
export async function GET() {
  try {
    const demos = listDemos()
    return NextResponse.json(createApiSuccess(demos))
  } catch (error) {
    return NextResponse.json(
      createApiError('FILE_READ_ERROR', '读取 Demo 列表失败'),
      { status: 500 }
    )
  }
}
```

### 4.2 参数验证

```typescript
export async function POST(request: NextRequest) {
  try {
    const body = await request.json()
    const { name } = body

    if (!name || typeof name !== 'string') {
      return NextResponse.json(
        createApiError('INVALID_REQUEST', 'name 参数必填'),
        { status: 400 }
      )
    }

    // ...
  } catch (error) {
    // ...
  }
}
```

### 4.3 资源检查

```typescript
export async function DELETE(
  _request: Request,
  { params }: { params: { id: string } }
) {
  try {
    const { id } = params

    if (!demoExists(id)) {
      return NextResponse.json(
        createApiError('DEMO_NOT_FOUND'),
        { status: 404 }
      )
    }

    deleteDemo(id)
    return NextResponse.json(createApiSuccess(null))
  } catch (error) {
    // ...
  }
}
```

```typescript
const { demos, isLoading, error } = useDemos()

if (error) {

```

### 5.2 API 调用错误处理

```typescript
async function createDemo(name: string) {
  try {
    const response = await fetch('/api/demos', {
      method: 'POST',
      body: JSON.stringify({ name }),
    })

    const result = await response.json()

    if (!result.success) {
      toast.error(result.error.message)
      return null
    }

    return result.data
  } catch (error) {
    toast.error('网络错误，请重试')
    return null
  }
}
```

### 6.2 错误详情

生产环境不应返回详细错误信息：

```typescript
export function createApiError(
  code: ErrorCodeType,
  message?: string,
  details?: unknown
) {
  return {
    success: false as const,
    error: {
      code,
      message: message || ERROR_MESSAGES[code],
      // 生产环境不返回 details
      ...(process.env.NODE_ENV !== 'production' && { details }),
    },
  }
}
```
