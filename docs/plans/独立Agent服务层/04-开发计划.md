export class AgentClient {
  private baseUrl: string;
  
  constructor(config: { baseUrl: string; apiKey?: string }) {
    this.baseUrl = config.baseUrl;
  }
  
  async sendMessage(sessionId: string, content: string): Promise<AgentResult> {
    const response = await fetch(`${this.baseUrl}/api/agent/${sessionId}/message`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ content }),
    });
    return response.json();
  }
  
  stream(sessionId: string): AgentStream {
    return new AgentStream(`${this.baseUrl}/api/agent/${sessionId}/stream`);
  }
}

const AGENT_SERVICE_URL = process.env.AGENT_SERVICE_URL || 'http://localhost:3001';

#### 任务 2.4：实现 OpenCodeBackend（新写，不参考 AionUi）

**注意**：AionUi 使用 stdio 与 CLI 进程通信，本项目使用 HTTP API，需要完全重写。

```typescript

import { Pool } from 'undici';

export class OpenCodeBackend extends BaseBackendAdapter {
  private pool: Pool;
  private sessionId?: string;

  async connect(): Promise<void> {
    this.pool = new Pool(this.serverUrl, {
      connections: 100,
      pipelining: 10,
    });
  }

  async sendMessage(content: string): Promise<AgentResult> {
    if (!this.sessionId) {
      this.sessionId = await this.createSession();
    }

    const response = await this.pool.request({
      method: 'POST',
      path: `/session/${this.sessionId}/message`,
      body: JSON.stringify({ parts: [{ type: 'text', text: content }] }),
    });

    return this.parseResponse(response);
  }
}
```

private async createOrResumeSession(): Promise<void> {
  const resumeSessionId = this.extra.acpSessionId;
  
  if (resumeSessionId) {
    try {
      // 尝试恢复已有会话
      const response = await this.connection.loadSession(resumeSessionId);
      if (response.sessionId !== resumeSessionId) {
        this.onSessionIdUpdate?.(response.sessionId);
      }
      return;
    } catch {
      // 恢复失败，创建新会话
    }
  }
  
  // 创建新会话
  const response = await this.connection.newSession();
  this.onSessionIdUpdate?.(response.sessionId);
}
